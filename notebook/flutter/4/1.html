# Flutter 发布

# 一、安卓发布

# 1. APP 名称与图标设置

APP名称：AndroidManifest.xml 修改

<application
        android:usesCleartextTraffic="true"
        android:name="io.flutter.app.FlutterApplication"
        android:label="优米社" -- App 名字

设置 snapshot 名字 （多任务显示名字）：main.dart 文件修改

MaterialApp(
  title: '优米社',

# 2. 检查和配置 build.gradle 文件

applicationId

// 位置在 app 的 build.gradle 中
// App的唯一标识
applicationId "com.umiclub.umiapp"
// 编译版本
compileSdkVersion 28 
// 发布版本
targetSdkVersion 28 
// 最小版本号
minSdkVersion 16

versionCode & versionName

// 版本号
versionCode flutterVersionCode.toInteger()
// 版本名
versionName flutterVersionName

# 3. App签名

签名 key

1）创建证书，build -> generate signed bundle / apk
2）填写 下列信息
文件名: umi_app_key_store
密码: 900629
Alias: umi_key
密码: 900629
3）创建 key.properties 文件：以 project 方式显示项目，在 android 目录下创建 Resource Bundle 类型文件
storePassword=900629
keyPassword=900629
keyAlias=umi_key
# 这个路径是以 android/app/build.gradle 文件的相对路径
storeFile=../umi_app_key_store
4）在 android/app/build.gradle 的 android { 节点上面 添加如下内容
// 设置 签名信息
def keystoreProperties = new Properties()
def keystorePropertiesFile = rootProject.file('key.properties')
if (keystorePropertiesFile.exists()) {
    keystoreProperties.load(new FileInputStream(keystorePropertiesFile))
}

android {
    // 编译版本
    compileSdkVersion 28
5）在 android 节点下，增加如下内容
andorid {
	```
	// 设置签名参数
    signingConfigs {
        release {
            keyAlias keystoreProperties['keyAlias']
            keyPassword keystoreProperties['keyPassword']
            storeFile file(keystoreProperties['storeFile'])
            storePassword keystoreProperties['storePassword']
        }
    }

    // 打包
    buildTypes {
        release {
            // TODO: Add your own signing config for the release build.
            // Signing with the debug keys for now, so `flutter run --release` works.
            // signingConfig signingConfigs.debug
            signingConfig signingConfigs.release
        }
    }

# 4. 打包命令

// 打包 64 位机子
flutter build apk --release --target-platform android-arm64
// 打包 32 位机子
flutter build apk --release --target-platform android-arm
